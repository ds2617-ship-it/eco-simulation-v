    let audioCtx;
    let bgmInterval;
    
    function initAudio() {
        if(audioCtx) return;
        audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        
        const notes = [
            1046.5, 1318.51, 1567.98, 1318.51, 
            1046.5, 1318.51, 1567.98, 0,       
            1174.66, 1396.91, 1760.00, 1396.91,
            1174.66, 1396.91, 1760.00, 0       
        ];
        
        let noteIndex = 0;
        bgmInterval = setInterval(() => {
            if(audioCtx.state === 'suspended') audioCtx.resume();
            const freq = notes[noteIndex];
            if(freq > 0) {
                const osc = audioCtx.createOscillator();
                const gain = audioCtx.createGain();
                osc.type = 'sine';
                osc.frequency.setValueAtTime(freq * 0.9, audioCtx.currentTime);
                osc.frequency.linearRampToValueAtTime(freq, audioCtx.currentTime + 0.05);
                gain.gain.setValueAtTime(0, audioCtx.currentTime);
                gain.gain.linearRampToValueAtTime(0.15, audioCtx.currentTime + 0.03); 
                gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.2);
                osc.connect(gain); gain.connect(audioCtx.destination);
                osc.start(); osc.stop(audioCtx.currentTime + 0.25);
            }
            noteIndex = (noteIndex + 1) % notes.length;
        }, 250);
    }

    function stopBGM() {
        if(bgmInterval) clearInterval(bgmInterval);
    }

    function playClickSound() {
        if(!audioCtx) return;
        const osc = audioCtx.createOscillator();
        const gain = audioCtx.createGain();
        osc.type = 'sine';
        osc.frequency.setValueAtTime(800, audioCtx.currentTime);
        osc.frequency.exponentialRampToValueAtTime(1200, audioCtx.currentTime + 0.1);
        gain.gain.setValueAtTime(0.15, audioCtx.currentTime);
        gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.15);
        osc.connect(gain); gain.connect(audioCtx.destination);
        osc.start(); osc.stop(audioCtx.currentTime + 0.15);
    }

    }

                        <span class="sphere-value"><span id="${key}-val">${s.value}</span>%</span>

                    <div class="progress-bar"><div id="${key}-bar" class="progress-fill" style="background-color:${s.color}; width:${Math.min(100, s.value)}%"></div></div>
                </div>`;
        }
    }

    function renderChoices() {
        const container = document.getElementById('choice-container');
        container.innerHTML = '';
        const pool = [...events].sort(() => Math.random() - 0.5).slice(0, 4); 
        pool.forEach(ev => {
            const btn = document.createElement('button');
            btn.className = 'choice-btn';
            btn.innerHTML = `
                <span class="icon-large">${ev.icon}</span>
                <div>
                    <div class="choice-title">${ev.title}</div>
                    <div class="choice-desc">${ev.desc}</div>
                </div>`;
            btn.onclick = () => { playClickSound(); selectEvent(ev); };
            container.appendChild(btn);
        });
    }

    function selectEvent(ev) {
        gameState.currentEv = ev;
        document.getElementById('stage-choose').style.display = 'none';
        document.getElementById('stage-predict').style.display = 'flex';
        document.getElementById('predict-header-title').innerText = ev.icon + " " + ev.title;
        document.getElementById('predict-header-desc').innerText = ev.desc;
        
        const pCon = document.getElementById('prediction-container');
        pCon.innerHTML = '<h4 style="margin-top:0; color:var(--accent); font-size: 1.5rem;">이 현상으로 인해 일어날 과학적 변화를 모두 고르세요 (함정 주의!)</h4>';
        
        [...ev.predictions].sort(() => Math.random() - 0.5).forEach(p => {
            const b = document.createElement('button');
            b.className = 'prediction-btn';
            b.innerHTML = `<span style="font-weight:bold; color:var(--accent);">[선택]</span> ${p.t}`;
            b.onclick = () => { playClickSound(); b.classList.toggle('selected'); };
            pCon.appendChild(b);
        });
    }

    function processTurn() {
        playTurnSound(); 
        
        const ev = gameState.currentEv;
        const selectedElements = Array.from(document.querySelectorAll('.prediction-btn.selected'));
        const selected = selectedElements.map(el => el.innerText.replace('[선택] ', '').trim());

        updateSphere(ev.direct.s, ev.direct.v);
        if(gameState.isGameOver) return;

        ev.predictions.forEach(p => {
            if (selected.includes(p.t)) {
                if (p.c) {
                    gameState.quizScore += 10;
                    if(p.a) for(const [k, v] of Object.entries(p.a)) updateSphere(k, v);
                } else {
                    gameState.quizScore -= 10; 
                }
            }
        });

        if(gameState.isGameOver) return;

        // 밸런스 체크 (생산자, 초식동물, 육식동물이 적절히 유지될 때)
        const total = spheres.producer.value + spheres.prey.value + spheres.predator.value;
        if(total >= 130 && total <= 220) updateSphere('biodiversity', 5);
        else updateSphere('biodiversity', -5);

        gameState.turn++;
        if (gameState.turn > gameState.maxTurns) endGame("축하합니다! 생태계의 복잡한 연쇄반응을 훌륭히 예측했습니다.");
        else {
            document.getElementById('stage-predict').style.display = 'none';
            document.getElementById('stage-choose').style.display = 'flex';
            document.getElementById('turn-count').innerText = `${gameState.turn} / 10 TURN`;
            document.getElementById('quiz-score-val').innerText = gameState.quizScore;
            renderChoices();
        }
    }

    function updateSphere(key, val) {
        // 생물 다양성은 상한선 없이 무한대로 증가 가능, 다른 수치는 최대 100 제한
        if (key === 'biodiversity') {
            spheres[key].value = Math.max(0, spheres[key].value + val);
        } else {
            spheres[key].value = Math.max(0, Math.min(100, spheres[key].value + val));
        }

        document.getElementById(`${key}-val`).innerText = spheres[key].value;
        
        // 막대그래프 시각적 길이는 100%까지만 차오르도록 제한하여 UI 깨짐 방지
        const barWidth = Math.min(100, spheres[key].value);
        document.getElementById(`${key}-bar`).style.width = barWidth + "%";
        
        if (spheres[key].value <= 0) {
            endGame(`[생태계 붕괴] ${spheres[key].name} 권역이 0%가 되어 멸종했습니다!`, true);
        }
    }

    function endGame(msg, isFailed = false) {
        gameState.isGameOver = true;
        stopBGM();
        
        document.getElementById('main-container').style.display = 'none';
        document.getElementById('game-header').style.display = 'none';
        const screen = document.getElementById('final-screen');
        screen.style.display = 'flex';
        document.getElementById('final-title').innerText = isFailed ? "GAME OVER" : "시뮬레이션 완수";
        document.getElementById('final-title').style.color = isFailed ? "var(--danger)" : "var(--success)";
        document.getElementById('final-msg').innerText = msg;
        document.getElementById('final-score-display').innerHTML = `
            최종 예측 점수: <span style="color:var(--warning);">${gameState.quizScore}점</span><br>
            <span style="font-size: 0.8em; color: #aaa;">최종 생물 다양성 달성률: <span style="color:var(--success); font-weight:bold;">${spheres.biodiversity.value}%</span></span>
        `;
    }
